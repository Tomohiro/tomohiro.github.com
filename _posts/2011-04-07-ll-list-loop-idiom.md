### リスト/配列の値を順番に評価する

PHP (Array)
{% highlight php %}
<?php
$l = array('foo', 'bar', 'baz');
foreach ($l as $v) {
    /* do something */
}
{% endhighlight %}

Perl (Array)
{% highlight perl %}
my @l = qw/foo bar baz/;
foreach my $v (@l) {
    # do something
}
{% endhighlight %}

Ruby (Array)
{% highlight ruby %}
l = %w[foo bar baz]
l.each do |v|
  # do something
end
{% endhighlight %}

Python (list)
{% highlight python %}
l = ['foo', 'bar', 'baz']
for v in l:
    # do something
{% endhighlight %}

JavaScript (Array)
{% highlight javascript %}
var l = ['foo', 'bar', 'baz'];
for (var i = 0; i < l.length; i++) {
    /* do something */
}
{% endhighlight %}

### リスト/配列のインデックスと値を順番に評価する

PHP (Array)
{% highlight php %}
<?php
$l = array('foo', 'bar', 'baz');
foreach ($l as $i => $v) {
    /* do something */
}
{% endhighlight %}

Perl (Array)
{% highlight perl %}
my @l = qw/foo bar baz/;
foreach my $i (0..$#l) {
    # do something
}
{% endhighlight %}

Ruby (Array)
{% highlight ruby %}
l = %w[foo bar baz]
l.each_with_index do |v, i|
  # do something
end
{% endhighlight %}

Python (list)
{% highlight python %}
l = ['foo', 'bar', 'baz']
for i, v in enumerate(l):
    # do something
{% endhighlight %}

JavaScript (Array)
{% highlight javascript %}
var l = ['foo', 'bar', 'baz'];
for (var i = 0; i < l.length; i++) {
    /* do something l[i] */
}
{% endhighlight %}
